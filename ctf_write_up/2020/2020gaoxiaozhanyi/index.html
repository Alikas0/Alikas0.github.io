from Crypto.Util.number import *

tab = b'ABCDEFGHJKLMNPQRSTUVWXYZ123456789abcdefghijkmnopqrstuvwxyz'

def b58decode(s):
    global tab
    tmp = b""
    for i in s[::-1]:
        tmp += bytes([tab.index(i)])
    pow = 1
    res = 0
    for i in range(len(tmp)):
        res += pow*tmp[i]
        pow *= 58
    return long_to_bytes(res)[:]
